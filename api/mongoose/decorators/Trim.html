Decorator Trim

# Decorator Trim

Decorator mongoose property
Module
import { Trim } from "@tsed/mongoose"
Source/packages/mongoose/src/decorators/trim.ts

# Overview

function Trim(): Function;

# Description

Tell Mongoose to make the property trimmable.

# Example

@Model()
export class EventModel {
  @Trim()
  name: string;
}
